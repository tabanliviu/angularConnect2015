Protractor Styleguide

Recommendations

1. Put all e2e test files in one dedicated folder

2. Use one configuration file + multiple tasks

3. XPATH

element(by.xpath("//*@id="calc"]/form/section/section[2]/div/div/div[1]/div/div/div[3]/div[2]/div[2]/div[1]/div[1]/h3"));

Never use XPATH

4. Page Objects

            
  /* avoid */

  describe('Question page', function() {
    it('should answer any question', function() {
      var question = element(by.model('question.text'));
      var answer = element(by.binding('answer'));
      var button = element(by.css('.question-button'));

      question.sendKeys("What is the purpose of meaning?");
      button.click();
      expect(answer.getText()).toEqual("Chocolate!");
    });
  });

4. Page Objects 

            
  /* recommended */

  /* question.page.js */
  var QuestionPage = function() {
    this.question = element(by.model('question.text'));
    this.answer = element(by.binding('answer'));
    this.button = element(by.className('question-button'));

    this.ask = function(question) {
      this.question.sendKeys(question);
      this.button.click();
    };
  };

  module.exports = UserPropertiesPage;

Protractor + ES6 generators

Generators are functions which can be exited and later re-entered. Their context (variable bindings) will be saved across re-entrances.

Before

            beforeEach(function() {
    http.put(DB_URL + '.json', cats)
      .then(function() {
        return browser.get('/index.html');
      })
      .then(function() {
        var mainElement = element(by.className('topcat-container'));
        return browser.wait(EC.presenceOf(mainElement));
      })
      .then(function() {
        return browser.wait(function () {
          return mainElement.evaluate('vm.ready');
        });
      });
  });

After

          
  beforeEach(co.wrap(function*() {
    yield http.put(DB_URL + '.json', cats);

    yield browser.get('/index.html');

    var mainElement = element(by.className('topcat-container'));
    yield browser.wait(EC.presenceOf(mainElement));
    yield browser.wait(function () {
      return mainElement.evaluate('vm.ready');
    });
  }));
